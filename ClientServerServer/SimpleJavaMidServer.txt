import java.io.InputStream;
import java.io.OutputStream;
import java.net.ServerSocket;
import java.net.Socket;

public class SimpleJavaMidServer {

    public static void main(String[] args) 	{
        try {
            ServerSocket midserver = new ServerSocket(6666);
            String str;
            while (true) {
                Socket cliente = midserver.accept();
                InputStream iCliente = cliente.getInputStream();
                OutputStream oCliente = cliente.getOutputStream();
                
                Socket server = new Socket("127.0.0.1", 9999);
                InputStream iServer = server.getInputStream();
                OutputStream oServer = server.getOutputStream();
                
                do {
                    byte[] line = new byte[100];
                    iCliente.read(line);
                    oServer.write(line);
                    iServer.read(line);
                    oCliente.write(line);
                    str = new String(line);
                } while ( !str.trim().equals("bye") );
                cliente.close();
                server.close();
            }
        }
        catch (Exception err){
            System.err.println(err);
        }
    }
}
