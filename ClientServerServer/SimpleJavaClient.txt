import java.io.InputStream;
import java.io.OutputStream;
import java.net.Socket;

public class SimpleJavaClient {

    public static void main(String[] args) 	{
        try {
            Socket midserver = new Socket("127.0.0.1", 6666);
            InputStream iMidServer = midserver.getInputStream();
            OutputStream oMidServer = midserver.getOutputStream();
            String str;
            do {
                byte[] line = new byte[100];
                System.in.read(line);
                oMidServer.write(line);
                iMidServer.read(line);
                str = new String(line);
                System.out.println(str.trim());
            } while ( !str.trim().equals("bye") );
            midserver.close();
        }
        catch (Exception err) {
            System.err.println(err);
        }
    }
}
